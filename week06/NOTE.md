有限状态机

```js
// 每个函数是一个状态
function state(input) {
  // 处理该状态的逻辑
  return next; // 返回下一个状态，next是固定的就是摩尔型状态机，和input有关就是米利型
}

// 调用
while(input) {
  // 获取输入
  state = state(input); // 把状态置为下一个状态
}
```
